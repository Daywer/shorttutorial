Оператор переключения Switch

Выполняет участок кода при котором значение переменной-аргумента равно метке.

 

switch ( x )        // где "x" переменная целого типа

{

    case  1:         //  если х == 1 выполняется одна или несколько строк кода

                          break;

    case  2:         //  если х == 2 выполняется одна или несколько строк кода

                          break;

    case  3:         //  если х == 3 выполняется одна или несколько строк кода

                          break;

    case  4:         //  если х == 4 выполняется одна или несколько строк кода

                          break;

    default:         //  если x не равно ни одному из выше перечисленного

}

 

оператор Switch работает не только с базовым типом данных. Ниже представлен пример для переменной строкового типа, которые реализован класcом String

switch ( string )        // где "string" переменная строкового типа

{

    case  "one":         //  если string == "one" выполняется одна или несколько строк кода

                                    break;

    case  "two":         //  если string == "two" выполняется одна или несколько строк кода

                                    break;

    case  "three":       //  если string == "three" выполняется одна или несколько строк кода

                                    break;

    default:                 //  если string не равно ни одному из выше перечисленного

}

 

У оператора  Switch есть измвестные особенности работы. Рекомендуем ознакомится. (см. SWITCH/BREAK)
